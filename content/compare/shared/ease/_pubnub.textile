var pubnub = new PubNub({
    publishKey: 'demo',
    subscribeKey: 'demo'
});

    channel: 'hello_world',
    message: {
        title: 'greeting',
        description: 'hello world!'
    }
};
pubnub.publish(publishConfig, function (status, response) {
    console.log(status, response)
});
